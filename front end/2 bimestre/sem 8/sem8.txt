// Importa o React e dois hooks importantes: useState (estado) e useEffect (efeitos colaterais)
import React, { useState, useEffect } from "react";

// Importa o arquivo de estilos CSS
import "./App.css";

// Componente principal da aplicação
function App() {

  // Estado que armazena a lista de tarefas (array)
  const [tasks, setTasks] = useState([]);

  // Estado que armazena o valor digitado no input
  const [input, setInput] = useState("");

  // Função para adicionar uma nova tarefa
  const addTask = () => {

    // Verifica se o input está vazio ou só com espaços
    if (input.trim() === "") return;

    // Cria um novo objeto de tarefa
    const newTask = { text: input, completed: false };

    // Atualiza o estado adicionando a nova tarefa ao array existente
    setTasks([...tasks, newTask]);

    // Limpa o campo de input
    setInput("");
  };

  // Função para marcar/desmarcar tarefa como concluída
  const toggleComplete = (index) => {

    // Cria um novo array de tarefas
    const newTasks = tasks.map((task, i) =>
      // Se o índice for o da tarefa clicada, altera o "completed"
      i === index ? { ...task, completed: !task.completed } : task,
    );

    // Atualiza o estado com o novo array
    setTasks(newTasks);
  };

  // Função para remover uma tarefa
  const removeTask = (index) => {

    // Filtra todas as tarefas, removendo a que tem o índice informado
    const newTasks = tasks.filter((_, i) => i !== index);

    // Atualiza o estado
    setTasks(newTasks);
  };

  // useEffect que salva as tarefas no localStorage sempre que "tasks" mudar
  useEffect(() => {
    localStorage.setItem("tasks", JSON.stringify(tasks));
  }, [tasks]);

  // useEffect que carrega as tarefas salvas quando o componente é iniciado
  useEffect(() => {

    // Recupera as tarefas do localStorage
    const savedTasks = JSON.parse(localStorage.getItem("tasks"));

    // Se existir algo salvo, atualiza o estado
    if (savedTasks) {
      setTasks(savedTasks);
    }
  }, []);

  // Retorno do componente (interface)
  return (
    <div className="App">

      {/* Título da aplicação */}
      <h1>To-Do List</h1>

      {/* Campo de entrada de texto */}
      <input
        type="text"
        value={input} // valor ligado ao estado
        onChange={(e) => setInput(e.target.value)} // atualiza o estado ao digitar
        placeholder="Nova tarefa"
      />

      {/* Botão que chama a função de adicionar tarefa */}
      <button onClick={addTask}>Adicionar</button>

      {/* Lista de tarefas */}
      <ul>
        {tasks.map((task, index) => (
          <li key={index} className="todo-item">

            {/* Texto da tarefa */}
            <span
              className={task.completed ? "completed" : ""} // aplica estilo se concluída
              onClick={() => toggleComplete(index)} // alterna estado ao clicar
            >
              {task.text}
            </span>

            {/* Botão para remover tarefa */}
            <button onClick={() => removeTask(index)}>Remover</button>
          </li>
        ))}
      </ul>
    </div>
  );
}

// Exporta o componente para ser usado em outros arquivos
export default App;
